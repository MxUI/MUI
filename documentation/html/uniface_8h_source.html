149  bool scan_spans_(time_type t, const span_t& s, const spans_type& spans ) const {
150  auto p = std::make_pair(t+threshold(t),t+threshold(t));
151  auto end = spans.upper_bound(p);
152  bool prefetched = false;
153 
154  for( auto itr = spans.begin(); itr != end; ++itr ) {
155  if( almost_equal(t, itr->first.second) || t < itr->first.second ) {
156  prefetched = true;
157  if( collide(s,itr->second) ) return true;
158  }
159  }
160  // if prefetched at t, but no overlap region, then return false;
161  // otherwise return true;
162  return !prefetched;
163  }
164  time_type latest_timestamp = std::numeric_limits<time_type>::lowest();
165  time_type next_timestamp = std::numeric_limits<time_type>::lowest();
166  spans_type recving_spans;
167  spans_type sending_spans;
168  std::vector<point_type> pts_;

300  template<class SAMPLER, class TIME_SAMPLER, typename ... ADDITIONAL>
301  typename SAMPLER::OTYPE
302  fetch( const std::string& attr,const point_type focus, const time_type t,
303  const SAMPLER& sampler, const TIME_SAMPLER &t_sampler,
304  bool barrier_enabled = true, time_type barrier_time = std::numeric_limits<time_type>::min(),
305  ADDITIONAL && ... additional ) {
306  if(barrier_enabled){
307  if(barrier_time == std::numeric_limits<time_type>::min())
308  barrier_time = t_sampler.get_upper_bound(t);
309  barrier(barrier_time);
310  }
311  std::vector<std::pair<time_type,typename SAMPLER::OTYPE> > v;
312 
313  for( auto first=log.lower_bound(t_sampler.get_lower_bound(t)-threshold(t)),
314  last = log.upper_bound(t_sampler.get_upper_bound(t)+threshold(t)); first!= last; ++first ){
315  time_type time = first->first;
316  const auto& iter = first->second.find(attr);
317  if( iter == first->second.end() ) continue;
318  v.emplace_back( time, iter->second.build_and_query_ts( focus, sampler, additional... ) );
319  }
320  return t_sampler.filter(t, v);
321  }

336  std::vector<point_type>
337  fetch_points( const std::string& attr, const time_type t, const TIME_SAMPLER &t_sampler,
338  bool barrier_enabled = true, time_type barrier_time = std::numeric_limits<time_type>::min(),
339  ADDITIONAL && ... additional ) {
340  using vec = std::vector<std::pair<point_type,TYPE> >;
341  if(barrier_enabled){
342  if(barrier_time == std::numeric_limits<time_type>::min())
343  barrier_time = t_sampler.get_upper_bound(t);

349  last = log.upper_bound(t_sampler.get_upper_bound(t)+threshold(t)); first != last; ++first ){
350  const auto& iter = first->second.find(attr);

355  return_points.emplace_back(ds[i].first);

391  int commit( time_type timestamp ) {
392  std::vector<bool> is_sending(comm->remote_size(), true);
393 
394  if( ((almost_equal(span_start, timestamp) || span_start < timestamp) && (almost_equal(timestamp, span_timeout) || timestamp < span_timeout)) ) {
395  for( std::size_t i=0; i<peers.size(); ++i ) {
396  is_sending[i] = peers[i].is_recving( timestamp, current_span );
397  }
398  }
399 
400  if( FIXEDPOINTS ) {
401  if( push_buffer_pts.size() > 0 ) {
402  comm->send(message::make("points",comm->local_rank(),std::move(push_buffer_pts)), is_sending);
403  initialized_pts_ = true;
404  }
405  comm->send(message::make("rawdata",comm->local_rank(),timestamp,std::move(push_buffer_raw)), is_sending);
406  push_buffer_raw.clear();
407  push_buffer_pts.clear();
408  }
409  else {
410  comm->send(message::make("data",timestamp,std::move(push_buffer)), is_sending);
411  push_buffer.clear();
412  }
413 
414  comm->send(message::make("timestamp",comm->local_rank(),timestamp));
415 
416  return std::count( is_sending.begin(), is_sending.end(), true );
417  }

422  bool is_ready( const std::string& attr, time_type t ) const {
423  using logitem_ref_t = typename decltype(log)::const_reference;
424  return std::any_of(log.begin(), log.end(), [=](logitem_ref_t time_frame) {
425  return time_frame.second.find(attr) != time_frame.second.end(); }) // return false for nonexisting attributes.
426  && std::all_of(peers.begin(), peers.end(), [=](const peer_state& p) {
427  return (!p.is_sending(t,recv_span)) || ((almost_equal(p.current_t(), t) || p.current_t() > t) || p.next_t() > t); });
428  }

430  void barrier( time_type t ) {
431  auto start = std::chrono::system_clock::now();
432  for(;;) { // barrier must be thread-safe because it is called in fetch()
433  std::lock_guard<std::mutex> lock(mutex);
434  if( std::all_of(peers.begin(), peers.end(), [=](const peer_state& p) {
435  return (!p.is_sending(t,recv_span)) || ((almost_equal(p.current_t(), t) || p.current_t() > t) || p.next_t() > t); }) ) break;
436  acquire(); // To avoid infinite-loop when synchronous communication
437  }
438  if( (std::chrono::system_clock::now() - start) > std::chrono::seconds(5) )
439  std::cerr << "MUI Warning [uniface.h]: Communication spends over 5 seconds" << std::endl;
440  }

499  void on_recv_data( time_type timestamp, frame_type frame ) {
500  // when message.id_ == "data"
501  auto first=log.lower_bound(timestamp-threshold(timestamp));
502  auto last=log.upper_bound(timestamp+threshold(timestamp));
503 
504  if( first == last ) {
505  if( last == log.end() ) std::tie(last,std::ignore) = log.insert(std::make_pair(timestamp,bin_frame_type()));
506  auto& cur = last->second;
507  for( auto& p: frame ){
508  auto pstr = cur.find(p.first);
509  if( pstr == cur.end() ) cur.insert(std::make_pair(std::move(p.first),spatial_t(std::move(p.second))));
510  else pstr->second.insert(p.second);
511  }
512  log.erase(log.begin(), log.upper_bound(timestamp-memory_length));
513  }
514  else {
515  size_t count=0;
516 
517  for( ; first != last; first++ ) {
518  count++;
519  }
520 
521  if( count > 1 )
522  std::cerr << "MUI Warning [uniface.h]: More than 1 time frame of data found in log for time=" << timestamp << std::endl;
523  }
524  }
