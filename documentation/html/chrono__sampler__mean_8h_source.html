67  template<typename TYPE>
68  TYPE filter( time_type focus, const std::vector<std::pair<time_type, TYPE> > &points ) const {
69  TYPE sum = TYPE(0);
70  for( auto i: points ) {
71  if ( i.first <= focus + right && i.first >= focus - left ) {
72  sum += i.second;
73  }
74  }
75  if ( points.size() ) return sum / TYPE(points.size());
76  else return TYPE(0);
77  }
78  time_type get_upper_bound( time_type focus ) const {
79  return focus + right;
80  }
81  time_type get_lower_bound( time_type focus ) const {
82  return focus - left;
83  }
84  time_type tolerance() const {
85  return time_type(0);
86  }
87 
88 protected:
89  time_type left, right;
