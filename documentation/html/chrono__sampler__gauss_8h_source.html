62  chrono_sampler_gauss( time_type newcutoff, REAL newsigma ) {
63  sigma = newsigma;
64  cutoff = newcutoff;
65  }
66 
67  template<typename TYPE>
68  TYPE filter( time_type focus, const std::vector<std::pair<time_type, TYPE> > &points ) const {
69  REAL wsum = REAL(0);
70  TYPE vsum = TYPE(0);
71  for( auto i: points ) {
72  time_type dt = std::abs(i.first - focus);
73  if ( dt < cutoff ) {
74  REAL w = pow( 2*PI*sigma, -0.5 ) * exp( -0.5 * dt * dt / sigma );
75  vsum += i.second * w;
76  wsum += w;
77  }
78  }
79  return ( wsum > std::numeric_limits<REAL>::epsilon() ) ? ( vsum / wsum ) : TYPE(0);
80  }
81  time_type get_upper_bound( time_type focus ) const {
82  return focus + cutoff;
83  }
84  time_type get_lower_bound( time_type focus ) const {
85  return focus - cutoff;
86  }
87  time_type tolerance() const {
88  return time_type(0);
89  }
90 
91 protected:
92  time_type cutoff;
93  REAL sigma;
94 };
