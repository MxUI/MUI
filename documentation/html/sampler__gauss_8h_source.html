64  sampler_gauss( REAL r_, REAL h_ ) : r(r_), h(h_), nh(std::pow(2*PI*h,-0.5*CONFIG::D)) {}
65 
66  template<template<typename,typename> class CONTAINER>
67  inline OTYPE filter( point_type focus, const CONTAINER<ITYPE,CONFIG> &data_points ) const {
68  REAL wsum = 0;
69  OTYPE vsum = 0;
70  for( size_t i = 0 ; i < data_points.size() ; i++ ) {
71  auto d = normsq( focus - data_points[i].first );
72  if ( d < r*r ) {
73  REAL w = nh * std::exp( (REAL(-0.5)/h) * d );
74  vsum += data_points[i].second * w;
75  wsum += w;
76  }
77  }
78  if ( wsum ) return vsum / wsum;
79  else return REAL(0.);
80  }
81 
82  inline geometry::any_shape<CONFIG> support( point_type focus, REAL domain_mag ) const {
83  return geometry::sphere<CONFIG>( focus, r );
84  }
85 
86 protected:
87  REAL r;
88  REAL h;
89  REAL nh;
