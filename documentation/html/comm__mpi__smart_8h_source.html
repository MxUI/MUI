67 private:
68  void send_impl_( message msg, const std::vector<bool> &is_sending ) {
69  test_completion();
70  auto bytes = std::make_shared<std::vector<char> >(msg.detach());
71  for( int i = 0 ; i < remote_size_ ; i++ )
72  if ( is_sending[i] ) {
73  send_buf.emplace_back(MPI_Request(), bytes);
74  MPI_Isend(bytes->data(), bytes->size(), MPI_BYTE, i, 0,
75  domain_remote_, &(send_buf.back().first));
76  }
77  }
78 
79  message recv_impl_() {
80  test_completion();
81  std::vector<char> temp;
82  MPI_Status status;
83  MPI_Probe(MPI_ANY_SOURCE, 0, domain_remote_, &status);
84  int count;
85  MPI_Get_count(&status,MPI_BYTE,&count);
86  temp.resize(count);
87  MPI_Recv( temp.data(), count, MPI_BYTE, status.MPI_SOURCE, status.MPI_TAG, domain_remote_, MPI_STATUS_IGNORE );
88  return message::make(std::move(temp));
89  }
90 
91  void test_completion() {
92  for( auto itr=send_buf.begin(), end=send_buf.end(); itr != end; ){
93  int test = false;
94  MPI_Test(&(itr->first),&test,MPI_STATUS_IGNORE);
95  if( test ) itr = send_buf.erase(itr);
96  else ++itr;
97  }
98  }
99 };
