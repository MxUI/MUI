67  sampler_sum_quintic( REAL r_ ) : r( r_ ), hinv( REAL( 3 ) / r_ )
68  {
69  static_assert( D == 1 || D == 2 || D == 3, "Quintic kernel for dimension other than 1,2,3 not defined." );
70  REAL sigma;
71  switch( D ) {
72  case 1:
73  sigma = 1.0 / 120.0;
74  break;
75  case 2:
76  sigma = 7.0 / 478.0 / PI;
77  break;
78  case 3:
79  sigma = 1.0 / 120.0 / PI;
80  break;
81  }
82  norm_factor = sigma * powr<D>( hinv );
83  }

85  template<template<typename, typename> class CONTAINER>
86  inline OTYPE filter( point_type focus, const CONTAINER<ITYPE, CONFIG> &data_points ) const
87  {
88  OTYPE vsum = 0;
89  for( size_t i = 0 ; i < data_points.size() ; i++ ) {
90  auto dist2 = normsq( focus - data_points[i].first );
91  if( dist2 < r * r ) {
92  REAL w = quintic_polynomial( sqrt( dist2 ) );
93  vsum += data_points[i].second * w;
94  }
95  }
96  return vsum;
97  }
98 
99  inline geometry::any_shape<CONFIG> support( point_type focus, REAL domain_mag ) const
100  {
101  return geometry::sphere<CONFIG>( focus, r );
102  }
103 
104 protected:
105  REAL r, hinv, norm_factor;

107  inline REAL quintic_polynomial( const REAL dist ) const
108  {
109  REAL s, w;
110  REAL s1, s2, s3;
111  REAL s1_5, s2_5, s3_5;
112  s = dist * hinv;
113  s1 = 1.0 - s;
114  s2 = 2.0 - s;
115  s3 = 3.0 - s;
116  s1_5 = 15.0 * powr<5>( s1 );
117  s2_5 = -6.0 * powr<5>( s2 );
118  s3_5 = powr<5>( s3 );
119  if( s < 1.0 ) {
120  w = s3_5 + s2_5 + s1_5;
121  } else if( s < 2.0 ) {
122  w = s3_5 + s2_5;
123  } else if( s < 3.0 ) {
124  w = s3_5;
125  } else {
126  w = 0.0;
127  }
128  w *= norm_factor;
129  return w;
130  }
131 };
