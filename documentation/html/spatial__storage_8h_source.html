100  void swap( spatial_storage& rhs ) noexcept(noexcept(BIN(std::move(rhs.bin_)))) {
101  data_.swap(rhs.data_);
102  if( is_bin_ && rhs.is_bin_ ) bin_.swap(rhs.bin_);
103  else if( is_bin_ && !rhs.is_bin_ ) {
104  ::new(&(rhs.bin_)) BIN(std::move(bin_));
105  rhs.is_bin_ = true;
106  destroy_if_bin_();
107  } else if( !is_bin_ && rhs.is_bin_ ) {
108  ::new(&(bin_)) BIN(std::move(rhs.bin_));
109  is_bin_ = true;

117  template<typename REGION, typename FOCUS, typename SAMPLER, typename ... ADDITIONAL>
118  typename SAMPLER::OTYPE
119  query(const REGION& reg, const FOCUS& f, const SAMPLER& s, ADDITIONAL && ... additional) const {
120  if( data_.empty() )
121  return s.filter( f, virtual_container<typename SAMPLER::ITYPE,CONFIG>(std::vector<std::pair<point_type,typename SAMPLER::ITYPE> >(),std::vector<bool>()), additional... );
122  if( !is_built() ) EXCEPTION(std::logic_error("MUI Error [spatial_storage.h]: Query error. "

148  void insert( storage_t storage ) {
149  destroy_if_bin_();

152  else if( data_.which() == storage.which() ) data_.apply_visitor(insert_{storage});
153  else EXCEPTION(bad_storage_id("MUI Error [spatial_storage.h]: Insert error. Type doesn't match."));
