219 template<typename CONFIG>
220 struct bin_t {
221 private:
222  using point_type = typename CONFIG::point_type;
223  static const int D = CONFIG::D;
224 
225  std::vector<std::size_t> displs;
226  // sorted[displs[i]] is the first element of the i-th bin
227 
228  point_type min, max;
229  std::size_t n[CONFIG::D];
230  typename CONFIG::REAL h;
231  using REAL = typename CONFIG::REAL;
232 public:
233  template<typename T>
234  bin_t( std::vector<std::pair<point_type,T> >& val ){
235  if( val.empty() ){
236  displs.resize(2,0);
237  h = 1.0;
238  return;
239  }
240  // calculate h & n
241  min = max = val[0].first;
242  for( std::size_t i=1; i<val.size(); ++i ){
243  point_type p = val[i].first;
244  for( std::size_t i=0; i<D; ++i ) {
245  min[i] = std::min(min[i],p[i]);
246  max[i] = std::max(max[i],p[i]);
247  }
248  }
249 
250  size_t zero_count=0;
251  REAL vol = abs(max[0]-min[0]);
252  if(almost_equal(vol, static_cast<REAL>(0))) { // check if first dimension is zero size, if so set to 1
253  vol = 1.0;
254  zero_count++;
255  }
256 
257  REAL vol_multi = vol;
258  for( int i=1; i<D; ++i ){
259  vol_multi = max[i]-min[i];
260  if(almost_equal(vol_multi, static_cast<REAL>(0))) { // check if other dimensions are zero size, if so set them to 1
261  vol_multi = static_cast<REAL>(1);
262  zero_count++;
263  }
264  vol *= vol_multi;
265  }
266 
267  if (zero_count == D) // if each dimension was actually zero (rather than just a subset) then set vol to zero
268  vol = static_cast<REAL>(0);
269 
270  h = std::pow(static_cast<REAL>(6)*vol/static_cast<REAL>(val.size()),1.0/D); // about 6 points per bin
271 
272  if(almost_equal(h, static_cast<REAL>(0))){ // if h is still zero (only in the case of all dimensions being zero) then warn the user as this may be a problem
273  h = static_cast<REAL>(1); // in this special case set h to 1 arbitrarily so bins work numerically
274  if(val.size() > 1)
275  std::cerr << "MUI Warning [bin.h]: Bin support size fixed to 1.0, check interface dimensionality or problem decomposition." << std::endl;
276  }
277 
278  std::size_t nn=1;
279  for( int i=0; i<D; ++i ) {
280  n[i] = static_cast<size_t>(std::ceil((max[i]-min[i])/h));
281  n[i] = static_cast<size_t>(std::max( n[i], std::size_t(1) ));
282  nn *= n[i];
283  }
284 
285  // make index
286  std::vector<std::size_t> index(val.size()+1,0);
287  std::vector<std::size_t> counts(nn,0);
288  for( std::size_t i=0; i<val.size(); ++i ) {
289  index[i] = get_index_(val[i].first);
290  ++counts[index[i]];
291  }
292  displs.resize(nn+1,0); // add 1 for sentinel
293  std::partial_sum(counts.begin(),counts.end(), displs.begin()+1);
294 
295  counts = displs;
296  std::vector<std::pair<point_type,T> > v(val.size());
297  for( std::size_t i=0; i<val.size(); ++i ) v[counts[index[i]]++] = val[i];
298  v.swap(val);
299  }

328  bool initialize_query_( const geometry::box<CONFIG>& bx, int lda[], int lh[][2] ) const {
329  bool broken = false;
330  lda[0] = 1;
331  for( int i=1; i<CONFIG::D; ++i ) lda[i] = lda[i-1]*n[i-1];
332  for( int i=0; i<CONFIG::D; ++i ) {
333  lh[i][0] = static_cast<int>(std::max(std::floor((bx.get_min()[i]-min[i])/h),typename CONFIG::REAL(0)));
334  lh[i][1] = static_cast<int>(std::min(std::ceil((bx.get_max()[i]-min[i])/h),typename CONFIG::REAL(n[i])));
335  if( lh[i][0] >= lh[i][1] ){
336  lh[i][0] = lh[i][1];
337  broken = true;
338  }
339  }
340  return broken;
341  }

343  inline std::size_t get_index_( const point_type& pt ) const {
344  std::size_t m = 1, ret=0;
345  for( int i=0; i<D; ++i ) {
346  std::size_t d = static_cast<size_t>((std::floor((pt[i]-min[i])/h)));
347  d = static_cast<size_t>(std::min(d,n[i]-1)); // the values may change here if (max[i]-min[i])/h is equal to a integer value.
348  ret += m*d;
349  m *= n[i];
350  }
351  return ret;
352  }
