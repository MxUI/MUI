73  void init( const char URI[], MPI_Comm world ) {
74  if ( initialized ) {
75  throw( std::runtime_error("MUI Error [comm_mpi.h]: Duplicate MUI communicator initialization") );
76  }
77 
78  // check MPI initialization status
79  int init;
80  MPI_Initialized( &init );
81  if (!init) {
82  init_by_me = true;
83  MPI_Init( NULL, NULL );
84  }
85 
86  // duplicate world to avoid potential conflict with host program
87  if (world == MPI_COMM_WORLD) MPI_Comm_dup( MPI_COMM_WORLD, &world );
88  MPI_Comm_size( world, &global_size_ );
89  MPI_Comm_rank( world, &global_rank_ );
90  if (global_size_ < 2) {
91  std::cerr << "MUI Warning [comm_mpi.h]: Number of global ranks less than 2" << std::endl;
92  }
93 
94  // get upper bond for tag hash
95  int prime;
96  {
97  int* v;
98  int flag;
99  MPI_Comm_get_attr(MPI_COMM_WORLD,MPI_TAG_UB,&v,&flag);
100  prime = *v;
101  }
102 
103  // parse URI, split the world using domain tag
104  uri desc(URI);
105  int domain_hash = std::hash<std::string>()( desc.host() );
106  domain_hash=std::abs(domain_hash);
107  int ifs_hash = std::hash<std::string>()( desc.path() );
108  ifs_hash=std::abs(ifs_hash);
109  MPI_Comm_split( world, domain_hash % prime, global_rank_, &domain_local_ );
110  MPI_Comm_size( domain_local_, &local_size_ );
111  MPI_Comm_rank( domain_local_, &local_rank_ );
112 
113  // set intercommunication using interface tag
114  std::vector<int> all_domain( global_size_, 0 );
115  std::vector<int> all_ifs ( global_size_, 0 );
116  MPI_Allgather( &domain_hash, 1, MPI_INT, &all_domain[0], 1, MPI_INT, world );
117  MPI_Allgather( &ifs_hash, 1, MPI_INT, &all_ifs[0], 1, MPI_INT, world );
118  for( int i = 0 ; i < global_size_ ; i++ ) {
119  if ( i == global_rank_ ) continue;
120  if ( all_domain[i] != domain_hash && all_ifs[i] == ifs_hash ) {
121  MPI_Intercomm_create( domain_local_, 0, world, i, ifs_hash % prime, &domain_remote_ );
122  break;
123  }
124  }
125  MPI_Comm_remote_size( domain_remote_, &remote_size_ );
126 
127  // output for debugging
128  for( int i=0;i<global_size_;i++ ) {
129  if ( i == global_rank_ )
130  std::cout <<"rank "<<global_rank_<<'\t'
131  <<"identifier "<<URI<<'\t'
132  <<"domain size "<<local_size_<<'\t'
133  <<"peer number "<<remote_size_<<'\t'
134  <<std::endl;
135  MPI_Barrier( world );
136  }
137 
138  initialized = true;
139  }
140 
141  void finalize() {
142  if ( initialized && init_by_me ) MPI_Finalize();
143  }
144

147  virtual int remote_size() const { return remote_size_; }
148  virtual int global_size() const { return global_size_; }
149  virtual int global_rank() const { return global_rank_; }
150

153  MPI_Comm domain_remote_;

157  int global_size_;

159 private:

161 };
